<?php
    if(isset($_POST["submit"])){
        //Name
        $nm = $_POST["name"];
        echo"Name: ".$nm;
        echo"<br><br>";

        //Roll No.
        $roll = $_POST["prn"];
        echo"Roll No.: ".$roll;
        echo"<br><br>";

        //Email
        $mail = $_POST["email"];
        if(filter_var($mail,FILTER_VALIDATE_EMAIL)){
            echo"Email: ".$mail;
        }
        echo"<br><br>";

        //Password
        $pass = $_POST["passwd"];
        echo"Password: ".$pass;
        echo"<br><br>";

        //Gender
        // $gen = $_POST["gender"];
        if(isset($_POST["gender"])){
            echo"Gender:".$_POST["gender"]."<br>";
        }

        //Phone No.
        $ph = $_POST["mobile"];
        if(preg_match("/^[0-9]{10}$/",$ph)){
            echo"Phone Number:" .$ph."<br>";
        }

        //Class
        $selected_val = $_POST['class'];  // Storing Selected Value In Variable
        echo "<br> You are in class :" .$selected_val;  // Displaying Selected Value

        //Courses
        if(isset($_POST['course'])){
            $a = $_POST['course'];
            echo "</br> You choose the following Technologies(s): <br>";
            foreach ($a as $course){
                echo $course."<br/>";
            }    
        }
        else{
            echo "<br> You did not choosen any Technology.";
        }
    }
?>
